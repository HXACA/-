CREATE TABLE miaosha_user(
	id BIGINT(20) NOT NULL,
	nickname VARCHAR(255) NOT NULL,
	PASSWORD VARCHAR(32) DEFAULT NULL,
	salt VARCHAR(10) DEFAULT NULL,
	head VARCHAR(128) DEFAULT NULL,
	register_date DATETIME DEFAULT NULL,
	last_login_date	DATETIME DEFAULT NULL,
	login_count INT(11) DEFAULT 0,
	PRIMARY KEY (id)
) ENGINE=INNODB DEFAULT CHARSET=utf8mb4ï¼›

CREATE TABLE goods(
  id BIGINT(20) NOT NULL AUTO_INCREMENT,
  goods_name VARCHAR(16) DEFAULT NULL,
  goods_title VARCHAR(64) DEFAULT NULL,
  goods_img VARBINARY(64) DEFAULT NULL,
  goods_detail LONGTEXT,
  goods_price DECIMAL(10,2) DEFAULT '0.00',
  goods_stock INT(11) DEFAULT '0',
  PRIMARY KEY(id)
)ENGINE=INNODB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4;

CREATE TABLE miaosha_goods(
  id BIGINT(20) NOT NULL AUTO_INCREMENT,
  goods_id BIGINT(20) DEFAULT NULL,
  miaosha_price DECIMAL(10,2) DEFAULT '0.00',
  stock_count INT(11) DEFAULT NULL,
  start_date DATETIME DEFAULT NULL,
  end_date DATETIME DEFAULT NULL,
  PRIMARY KEY(id)
)ENGINE=INNODB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4;

CREATE TABLE order_info(
  id BIGINT(20) NOT NULL AUTO_INCREMENT,
  user_id BIGINT(20) DEFAULT NULL,
  goods_id BIGINT(20) DEFAULT NULL,
  delivery_addr_id BIGINT(20) DEFAULT NULL,
  goods_name VARCHAR(16) DEFAULT NULL,
  goods_account INT(11) DEFAULT '0',
  goods_price DECIMAL(10,2) DEFAULT '0.00',
  order_channel TINYINT(4) DEFAULT '0',
  STATUS TINYINT(4) DEFAULT '0',
  create_date DATETIME	DEFAULT NULL,
  pay_date DATETIME DEFAULT NULL,
  PRIMARY KEY(id)
)ENGINE=INNODB AUTO_INCREMENT=12 DEFAULT CHARSET=utf8mb4;

CREATE TABLE miaosha_order(
  id BIGINT(20) NOT NULL AUTO_INCREMENT,
  user_id BIGINT(20) DEFAULT NULL,
  order_id BIGINT(20) DEFAULT NULL,
  goods_id BIGINT(20) DEFAULT NULL,
  PRIMARY KEY(id)
)ENGINE=INNODB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4;
